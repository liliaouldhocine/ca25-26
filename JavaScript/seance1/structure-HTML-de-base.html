Exploration de l'objet Window en JavaScript

Cette page contient une série d'exercices pour explorer et comprendre l'objet Window, les éléments du DOM et leurs attributs.

Exercice 1: Propriétés de la fenêtre

Affichez les propriétés suivantes de l'objet window dans la zone de résultat :

  • La largeur et la hauteur de la fenêtre (innerWidth et innerHeight)
  • Les coordonnées de la fenêtre par rapport à l'écran (screenX et screenY)
  • L'URL de la page actuelle (location.href)
function exercise1() {
    const result = document.getElementById('result1');
    result.innerHTML = `
        Largeur de la fenêtre: ${window.innerWidth}px

        Hauteur de la fenêtre: ${window.innerHeight}px

        Position X par rapport à l'écran: ${window.screenX}

        Position Y par rapport à l'écran: ${window.screenY}

        URL actuelle: ${window.location.href}
    `;
}

Exercice 2: Méthodes de navigation

}

function reloadPage() {
    window.location.reload();
}

function goBack() {
    window.history.back();
}

function goForward() {
    window.history.forward();
}

Exercice 3: Timers et intervalles

Créez un minuteur qui :

  • Affiche un compte à rebours de 10 secondes
  • Affiche un message quand le temps est écoulé
  • Permet de démarrer,暂停 et réinitialiser le minuteur

10

let timerValue = 10;
let timerInterval;

function startTimer() {
    clearInterval(timerInterval);
    timerInterval = setInterval(() => {
        if (timerValue > 0) {
            timerValue--;
            document.getElementById('timer').textContent = timerValue;
        } else {
            clearInterval(timerInterval);
            alert("Temps écoulé !");
        }
    }, 1000);
}

function pauseTimer() {
    clearInterval(timerInterval);
}

function resetTimer() {
    clearInterval(timerInterval);
    timerValue = 10;
    document.getElementById('timer').textContent = timerValue;
}

  • Ajoute et supprime des classes CSS

Ceci est un texte à modifier.

function changeText() {
    const element = document.getElementById('text-to-change');
    element.textContent = "Le texte a été modifié avec succès!";
}

function changeStyle() {
    const element = document.getElementById('text-to-change');
    element.style.color = 'red';
    element.style.fontSize = '20px';
}

function toggleClass() {
    const element = document.getElementById('text-to-change');
    element.classList.toggle('highlight');
}

Créez des gestionnaires d'événements pour :

  • Détecter le redimensionnement de la fenêtre
  • Détecter le défilement de la page
  • Afficher les coordonnées de la souris lorsqu'elle se déplace

// Au chargement de la page
window.addEventListener('load', () => {
    // Événement de redimensionnement
    window.addEventListener('resize', () => {
        document.getElementById('window-size').textContent = 
            `Taille de la fenêtre: ${window.innerWidth}x${window.innerHeight}`;
    });
    
    // Événement de défilement
    window.addEventListener('scroll', () => {
        document.getElementById('scroll-position').textContent = 
            `Position de défilement: ${window.scrollY}px`;
    });
    
    // Événement de mouvement de souris
    document.addEventListener('mousemove', (e) => {
        document.getElementById('mouse-position').textContent = 
            `Position de la souris: X=${e.clientX}, Y=${e.clientY}`;
    });
});
